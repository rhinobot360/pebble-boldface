// Phone-side code: settings + weather from Open-Meteo var SETTINGS_URL = 'https://rhinobot360.github.io/pebble-boldface/'; var ANALYTICS_URL = 'https://script.google.com/macros/s/AKfycbzUpTpx8q_fy3Ppg780g-RjekdlS0imaF5CuKJ_bFQQUrei7eSkZdGl0pXm-Dt7Z2TxTg/exec'; var APP_VERSION = '1.0'; var xhrRequest = function(url, type, callback) { var xhr = new XMLHttpRequest(); xhr.onload = function() { callback(this.responseText); }; xhr.open(type, url); xhr.send(); }; function pingUsage() { var now = Date.now(); var lastPing = parseInt(localStorage.getItem('last_ping') || '0'); var oneDay = 24 * 60 * 60 * 1000; if (now - lastPing < oneDay) return; // Already pinged today var THEME_LABELS = ['Tangerine','Blueberry','Lime','Licorice','Avocado','Cherry','Grape','Pink Lemonade']; var VIS_LABELS = ['always_on','shake_to_show','off']; var themeIndex = parseInt(localStorage.getItem('theme') || '0'); var visIndex = parseInt(localStorage.getItem('visibility') || '0'); var params = [ 'theme=' + encodeURIComponent(THEME_LABELS[themeIndex] || 'Tangerine'), 'weather=' + (localStorage.getItem('show_weather') === '0' ? 'off' : 'on'), 'date=' + (localStorage.getItem('show_date') === '0' ? 'off' : 'on'), 'steps=' + (localStorage.getItem('show_steps') === '0' ? 'off' : 'on'), 'visibility=' + encodeURIComponent(VIS_LABELS[visIndex] || 'always_on'), 'device_id=' + encodeURIComponent(localStorage.getItem('device_id') || 'unknown'), 'version=' + APP_VERSION ].join('&'); xhrRequest(ANALYTICS_URL + '?' + params, 'GET', function(response) { console.log('Usage ping sent: ' + response); localStorage.setItem('last_ping', now.toString()); }); } function locationSuccess(pos) { var lat = pos.coords.latitude; var lon = pos.coords.longitude; var useCelsius = localStorage.getItem('use_celsius') === '1'; var url = 'https://api.open-meteo.com/v1/forecast' + '?latitude=' + lat + '&longitude=' + lon + '¤t=temperature_2m,weathercode' + '&temperature_unit=' + (useCelsius ? 'celsius' : 'fahrenheit'); xhrRequest(url, 'GET', function(responseText) { var json = JSON.parse(responseText); var temp = Math.round(json.current.temperature_2m); var code = json.current.weathercode; var icon = 0; if (code === 0) icon = 0; // Clear else if (code <= 2) icon = 1; // Partly cloudy else if (code === 3) icon = 2; // Cloudy else if (code <= 49) icon = 6; // Fog else if (code <= 67) icon = 3; // Rain else if (code <= 77) icon = 5; // Snow else if (code <= 82) icon = 3; // Rain showers else if (code <= 99) icon = 4; // Thunder Pebble.sendAppMessage({ 'KEY_TEMPERATURE': temp, 'KEY_WEATHER_ICON': icon, 'KEY_USE_CELSIUS': useCelsius ? 1 : 0 }, function() { console.log('Weather sent to watch'); }, function(e) { console.log('Weather send failed: ' + JSON.stringify(e)); }); }); } function locationError(err) { console.log('Location error: ' + err); } function getWeather() { navigator.geolocation.getCurrentPosition(locationSuccess, locationError, { timeout: 15000, maximumAge: 60000 }); } // Open settings page Pebble.addEventListener('showConfiguration', function() { var theme = localStorage.getItem('theme') || '0'; var show_weather = localStorage.getItem('show_weather') || '1'; var show_date = localStorage.getItem('show_date') || '1'; var show_steps = localStorage.getItem('show_steps') || '1'; var use_celsius = localStorage.getItem('use_celsius') || '0'; var url = SETTINGS_URL + '?theme=' + theme + '&show_weather=' + show_weather + '&show_date=' + show_date + '&show_steps=' + show_steps + '&use_celsius=' + use_celsius + Pebble.openURL(url); }); // Receive device ID from watch Pebble.addEventListener('appmessage', function(e) { if (e.payload.KEY_DEVICE_ID) { localStorage.setItem('device_id', e.payload.KEY_DEVICE_ID); } }); // Handle settings return Pebble.addEventListener('webviewclosed', function(e) { if (e.response && e.response !== 'CANCELLED') { var settings = JSON.parse(decodeURIComponent(e.response)); // Save to localStorage localStorage.setItem('theme', settings.KEY_THEME); localStorage.setItem('show_weather', settings.KEY_SHOW_WEATHER); localStorage.setItem('show_date', settings.KEY_SHOW_DATE); localStorage.setItem('show_steps', settings.KEY_SHOW_STEPS); localStorage.setItem('use_celsius', settings.KEY_USE_CELSIUS); localStorage.setItem('visibility', settings.KEY_VISIBILITY); // Send all settings to watch Pebble.sendAppMessage({ 'KEY_THEME': settings.KEY_THEME, 'KEY_SHOW_WEATHER': settings.KEY_SHOW_WEATHER, 'KEY_SHOW_DATE': settings.KEY_SHOW_DATE, 'KEY_SHOW_STEPS': settings.KEY_SHOW_STEPS, 'KEY_USE_CELSIUS': settings.KEY_USE_CELSIUS, 'KEY_VISIBILITY': settings.KEY_VISIBILITY }, function() { console.log('Settings sent to watch'); getWeather(); }, function(e) { console.log('Settings send failed: ' + JSON.stringify(e)); }); } }); // Fetch weather on launch and every 30 minutes Pebble.addEventListener('ready', function() { getWeather(); pingUsage(); setInterval(getWeather, 30 * 60 * 1000); }); // Fetch weather when watch requests it Pebble.addEventListener('appmessage', function(e) { getWeather(); });
